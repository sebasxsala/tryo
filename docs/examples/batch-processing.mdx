---
title: Batch Processing
description: Execute multiple tasks with limits and result aggregation
---

## Example

```typescript
import { runAllSettled, isSuccess } from "runtry";

const tasks = Array.from({ length: 20 }, (_, i) => async () => i * 2);

const results = await runAllSettled<number>(tasks, { concurrency: 5, mode: "settle" });

const values = results.filter(isSuccess).map((r) => r.data);
```
