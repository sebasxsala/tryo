## `run` para una tarea

```typescript
import { run } from "trybox";

const r = await run(() => fetch("/api").then((x) => x.json()));
if (r.ok) console.log(r.data);
else console.error(r.error.code);
```

## `runAll` para varias tareas

```typescript
import { runAll, isSuccess } from "trybox";

const tasks = [() => fetch("/a"), () => fetch("/b")];
const results = await runAll<Response>(tasks, { concurrency: 2 });

const successes = results.filter(isSuccess).map((r) => r.data);
```

## Reusar un runner

```typescript
import trybox from "trybox";
export const runner = trybox({ ignoreAbort: true });

const r = await runner.run(() => fetch("/api"));
```
