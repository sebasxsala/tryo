## De try/catch a `run`

```typescript
import { run } from "tryo";

const r = await run(() => fetch("/api").then((x) => x.json()));
if (r.ok) {
  console.log(r.data);
} else {
  console.error(r.error.code);
}
```

## De `Promise.all` a `all`

```typescript
import { all } from "tryo";

const tasks = [() => fetch("/a"), () => fetch("/b")];
const results = await all<Response>(tasks, {
  concurrency: 2,
  mode: "settle",
});
```
